\caption{Option vocabulary. These inference-rule macros cover the proof patterns in PrOntoQA.}
\label{tab:options}
\begin{tabular}{llll}
\hline
\textbf{Option} & \textbf{Sym.} & \textbf{Args} & \textbf{Rule} \\
\hline
\texttt{MODUS\_PONENS} & MP & $(i, j)$ & $P, P{\rightarrow}Q \vdash Q$ \\
\texttt{MODUS\_TOLLENS} & MT & $(i, j)$ & $\neg Q, P{\rightarrow}Q \vdash \neg P$ \\
\texttt{UNIV\_INST} & UI & $(i, c)$ & $\forall x.P(x) \vdash P(c)$ \\
\texttt{EXIST\_GEN} & EG & $(i, x)$ & $P(c) \vdash \exists x.P(x)$ \\
\texttt{AND\_INTRO} & $\land$I & $(i, j)$ & $P, Q \vdash P \land Q$ \\
\texttt{AND\_ELIM} & $\land$E & $(i, s)$ & $P \land Q \vdash P$ or $Q$ \\
\texttt{OR\_INTRO} & $\lor$I & $(i, Q)$ & $P \vdash P \lor Q$ \\
\texttt{DISJ\_SYLL} & DS & $(i, j)$ & $P{\lor}Q, \neg P \vdash Q$ \\
\texttt{HYPO\_SYLL} & HS & $(i, j)$ & $P{\rightarrow}Q, Q{\rightarrow}R \vdash P{\rightarrow}R$ \\
\texttt{DOUBLE\_NEG} & DN & $(i)$ & $\neg\neg P \vdash P$ \\
\texttt{CONCLUDE} & --- & $(l)$ & \parbox{2.5cm}{Terminal:\\ T/F/Unknown} \\
\hline
\end{tabular}
